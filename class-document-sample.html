Create new application on Insellerate.

            
                <?php
                use BeSmartee\InsellerateSdk;

                $client = new InsellerateClient('https://stg.insellerate.com');
                $orgId = 112345; // TODO: Replace with your org id
                $campaignId = 1101; // TODO: Replace with your campaign id
                //
                /**
                 * Refer to:
                 * @link: https://app.insellerate.com/Help/Api/POST-api-integration-CampaignPost-orgId-campaignId
                 */
                $postData = [
                    'root' => [
                        'row' => [
                            [
                                'Ref_ID'        => 'PUBLICID',
                                'First_Name'    => 'Alice',
                                'Last_Name'     => 'Firstimer',
                                'Middle_Name'   => '',
                                'Email'         => 'alice@besmartee.com',
                                'SSN'           => '991919991',
                                'Phone'         => '8888888888',
                                'DateOfBirth'   => '10/01/1993',
                            ]
                        ]
                    ]
                ];
                $result = $client->integrationCampaign->campaignPost($orgId, $campaignId, $postData);

                if ($result !== false && $result['StatusCode'] == 1) {
                    // Successful
                    $prospectId = $result['Data']['ProspectId'];
                    $applicationId = $result['Data']['ApplicationId'];
                    // do something with $prospectId & $applicationId
                }
                else {
                    // Handle error
                }

Get application fields of client.

            
                <?php
                use BeSmartee\InsellerateSdk;

                $client = new InsellerateClient('https://stg.insellerate.com');

                $orgId = 112345; // TODO: Replace with your org id
                // @link: https://app.insellerate.com/Help/Api/GET-api-integration-GetApplicationFields_orgId
                $result = $client->integrationSystemLookup->getApplicationFields($orgId);

                if ($result !== false && $result['StatusCode'] == 1) {
                    // Successful
                    $fields = $result['Data'];
                    var_dump($fields);
                }
                else {
                    echo 'Error occurs';
                }
